# --- GLFW detection macro (non-Windows; deduplicates macOS and Linux detection) ---
macro(find_or_fetch_glfw)
    set(GLFW_FOUND_SYSTEM FALSE)
    find_package(glfw3 3.3 QUIET)
    if(glfw3_FOUND)
        set(GLFW_FOUND_SYSTEM TRUE)
        message(STATUS "Found system GLFW via find_package")
    else()
        find_package(PkgConfig QUIET)
        if(PkgConfig_FOUND)
            pkg_check_modules(GLFW3 QUIET glfw3>=3.3)
            if(GLFW3_FOUND)
                set(GLFW_FOUND_SYSTEM TRUE)
                message(STATUS "Found system GLFW via pkg-config: ${GLFW3_VERSION}")
            endif()
        endif()
    endif()
    if(NOT GLFW_FOUND_SYSTEM)
        message(STATUS "System GLFW 3.3+ not found - downloading and building GLFW 3.4 via FetchContent")
        message(STATUS "  (This may take a moment on first configure)")
        message(STATUS "  Alternatively, install via: brew install glfw")
        include(FetchContent)
        set(GLFW_BUILD_DOCS OFF CACHE BOOL "" FORCE)
        set(GLFW_BUILD_TESTS OFF CACHE BOOL "" FORCE)
        set(GLFW_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
        set(GLFW_INSTALL OFF CACHE BOOL "" FORCE)
        if(APPLE)
            # Wayland is not applicable on macOS
            set(GLFW_BUILD_WAYLAND OFF CACHE BOOL "" FORCE)
        endif()
        FetchContent_Declare(
            glfw
            GIT_REPOSITORY https://github.com/glfw/glfw.git
            GIT_TAG        3.4
            GIT_SHALLOW    TRUE
            GIT_PROGRESS   TRUE
        )
        FetchContent_MakeAvailable(glfw)
        message(STATUS "GLFW 3.4 will be built from source")
    endif()
endmacro()

# Link GLFW to a target (call after find_or_fetch_glfw)
function(link_glfw target_name)
    if(glfw3_FOUND)
        target_link_libraries(${target_name} PRIVATE glfw)
    elseif(GLFW3_FOUND)
        target_link_libraries(${target_name} PRIVATE ${GLFW3_LIBRARIES})
        target_include_directories(${target_name} PRIVATE ${GLFW3_INCLUDE_DIRS})
    else()
        # FetchContent GLFW - target 'glfw' created by FetchContent_MakeAvailable
        target_link_libraries(${target_name} PRIVATE glfw)
    endif()
endfunction()

# FAST_LIBS_BOOST option: Choose hash map implementation
#   - FAST_LIBS_BOOST=ON: Use boost::unordered_map + boost::unordered_flat_map (requires Boost 1.81+)
#     boost::unordered_flat_map (added in 1.81) is open-addressing with inline value storage,
#     providing cache-friendly iteration over large maps (e.g. id_to_entry_ with 1M+ entries).
#   - Default: Use std::unordered_map (standard library, works everywhere)
option(FAST_LIBS_BOOST "Use Boost unordered containers for faster hash maps (requires Boost 1.81+)" OFF)

if(FAST_LIBS_BOOST)
    if(POLICY CMP0144)
        cmake_policy(SET CMP0144 NEW)
    endif()
    if(POLICY CMP0167)
        cmake_policy(SET CMP0167 NEW)
    endif()
    find_package(Boost 1.81.0 REQUIRED)
    if(Boost_FOUND)
        target_compile_definitions(find_helper PRIVATE FAST_LIBS_BOOST)
        if(TARGET Boost::boost)
            target_link_libraries(find_helper PRIVATE Boost::boost)
        else()
            target_include_directories(find_helper PRIVATE ${Boost_INCLUDE_DIRS})
        endif()
        message(STATUS "FAST_LIBS_BOOST enabled: Using boost::unordered_map + boost::unordered_flat_map")
        message(STATUS "  Boost version: ${Boost_VERSION}")
        if(Boost_INCLUDE_DIRS)
            message(STATUS "  Boost include dirs: ${Boost_INCLUDE_DIRS}")
        endif()
    else()
        message(FATAL_ERROR "FAST_LIBS_BOOST requires Boost 1.81+ but Boost was not found.")
    endif()
else()
    message(STATUS "Using standard library hash maps (std::unordered_map)")
endif()

# ASan helper function
function(apply_asan_to_target target_name)
    if(ENABLE_ASAN AND NOT ENABLE_TSAN AND CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
        target_compile_options(${target_name} PRIVATE -fsanitize=address -fno-omit-frame-pointer)
        target_link_options(${target_name} PRIVATE -fsanitize=address)
    endif()
endfunction()

# TSAN helper function (macOS/Linux only; not supported on Windows - see LLVM ThreadSanitizer docs).
function(apply_tsan_to_target target_name)
    if(ENABLE_TSAN AND NOT MSVC AND CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
        target_compile_options(${target_name} PRIVATE -fsanitize=thread -fno-omit-frame-pointer)
        target_link_options(${target_name} PRIVATE -fsanitize=thread)
    endif()
endfunction()

# Coverage helper function
function(apply_coverage_to_target target_name)
    if(ENABLE_COVERAGE AND CMAKE_CXX_COMPILER_ID MATCHES "Clang")
        target_compile_options(${target_name} PRIVATE
            -fprofile-instr-generate
            -fcoverage-mapping
        )
        target_link_options(${target_name} PRIVATE
            -fprofile-instr-generate
            -fcoverage-mapping
        )
    elseif(ENABLE_COVERAGE AND CMAKE_CXX_COMPILER_ID MATCHES "GNU")
        target_compile_options(${target_name} PRIVATE
            --coverage
            -fprofile-arcs
            -ftest-coverage
        )
        target_link_options(${target_name} PRIVATE --coverage)
    endif()
endfunction()

# Apply MSVC /LTCG and optional PGO /GL /Gy to a test target that links /GL object libraries.
# Only add /LTCG when ENABLE_PGO is true (tests get /GL); otherwise the linker warns
# "no code generation required" and suggests removing /LTCG for performance.
function(configure_msvc_ltcg target_name)
    if(MSVC)
        if(ENABLE_PGO)
            target_compile_options(${target_name} PRIVATE
                $<$<CONFIG:Release>:/GL /Gy>
            )
            target_link_options(${target_name} PRIVATE
                $<$<CONFIG:Release>:/LTCG /OPT:REF /OPT:ICF>
            )
        endif()
    endif()
endfunction()

# Configure a test target with standard compiler options, USN_WINDOWS_TESTS, and common includes.
# Optional keyword: WITH_AVX2 — enables -mavx2 on x86_64 GCC/Clang builds.
function(configure_test_target target_name)
    cmake_parse_arguments(ARG "WITH_AVX2" "" "" ${ARGN})

    target_compile_features(${target_name} PRIVATE cxx_std_17)

    if(MSVC)
        target_compile_options(${target_name} PRIVATE
            $<$<CONFIG:Debug>:/Zi /Od>
            $<$<CONFIG:Release>:/O2>
        )
    elseif(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
        if(ARG_WITH_AVX2 AND CMAKE_SYSTEM_PROCESSOR MATCHES "(x86_64|AMD64|i[3-6]86)")
            target_compile_options(${target_name} PRIVATE
                $<$<CONFIG:Debug>:-g -O0 -mavx2>
                $<$<CONFIG:Release>:-O2 -mavx2>
            )
        else()
            target_compile_options(${target_name} PRIVATE
                $<$<CONFIG:Debug>:-g -O0>
                $<$<CONFIG:Release>:-O2>
            )
        endif()
    endif()

    if(NOT WIN32 AND CROSS_PLATFORM_TESTS)
        target_compile_definitions(${target_name} PRIVATE USN_WINDOWS_TESTS)
    endif()

    target_include_directories(${target_name} PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/src
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${DOCTEST_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}/external/nlohmann_json/single_include
    )
endfunction()
